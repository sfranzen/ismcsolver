99  void iterate(TreeMap &trees, const Game<Move> &state) const
100  {
101  const auto iterations = this->iterationCount();
102  if (iterations > 0) {
103  for (std::size_t i {0}; i < iterations; ++i)

117  void search(TreeMap &trees, const Game<Move> &rootState) const
118  {
119  NodePtrMap roots;
120  for (auto &pair : trees)
121  roots.emplace(pair.first, pair.second.get());
122  auto randomState = rootState.cloneAndRandomise(rootState.currentPlayer());
123  select(roots, *randomState);
124  expand(roots, *randomState);
125  this->simulate(*randomState);
126  backPropagate(roots, *randomState);
127  }

135  void select(NodePtrMap &nodes, Game<Move> &state) const
136  {
137  const auto validMoves = state.validMoves();
138  const auto player = state.currentPlayer();
139  const auto &targetNode = nodes[player];
140  if (!MOSolver::selectNode(targetNode, validMoves)) {
141  const auto selection = this->selectChild(targetNode, state, validMoves);
142  for (auto &node : nodes)
143  node.second = MOSolver::findOrAddChild(node.second, state, selection->move());
144  state.doMove(selection->move());
145  select(nodes, state);
146  }
