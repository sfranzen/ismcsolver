74  Node *selectChild(const std::vector<Move> &legalMoves, const TreePolicy<Node> &policy) const
75  {
76  std::vector<Node*> legalChildren;
77  legalChildren.reserve(m_children.size());
78  for (const auto &node : m_children) {
79  if (std::find(legalMoves.begin(), legalMoves.end(), node->m_move) != legalMoves.end()) {
80  legalChildren.emplace_back(node.get());
81  ++(node->m_available);
